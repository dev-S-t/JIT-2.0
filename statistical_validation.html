Statistical Validation: Multi-Run Simulation Analysis¶

30-Iteration Monte Carlo Simulation with Confidence Intervals & Significance Tests¶

Purpose: A single simulation run can be influenced by a particular random seed — the specific demand pattern, trauma events, and prediction noise of that run. To establish statistical robustness, we repeat the full pipeline (data generation → model training → inventory simulation) across 30 independent iterations with different random seeds.

For each iteration we record the wastage and shortage rates of all three strategies, then compute:

  • Mean and standard deviation across runs
  • 95% confidence intervals (Student's t)
  • Paired t-tests (p-values) to confirm the improvements are statistically significant

from datetime import datetime, timedelta

SCENARIOS = {
    "Hamilton_Medium_Hospital": {"mean_demand": 17.9, "std_demand": 7.05}
}
START_DATE, END_DATE = datetime(2024,1,1), datetime(2025,12,31)

WEEKDAY_MULT = {0:1.10, 1:1.15, 2:1.05, 3:1.05, 4:1.20, 5:0.70, 6:0.70}
MONTH_MULT = {1:0.90,2:0.95,3:1.00,4:1.00,5:1.02,6:1.05,7:1.08,8:1.05,9:1.02,10:1.00,11:0.98,12:0.88}

HOLIDAYS = set()
for y in [2024, 2025]:
    for m, d in [(1,1),(1,15),(2,19),(5,27),(7,4),(9,2),(11,28),(11,29),(12,24),(12,25),(12,31)]:
        try:
            HOLIDAYS.add(datetime(y, m, d).date())
        except:
            pass

def generate_data(seed):
    np.random.seed(seed); random.seed(seed)
    mean, std = 17.9, 7.05
    var = std**2; p = mean/var; n = mean*p/(1-p)
    dates = pd.date_range(START_DATE, END_DATE, freq='D')
    rows = []
    for date in dates:
        base = sp_stats.nbinom.rvs(n, p)
        wm = WEEKDAY_MULT[date.weekday()]
        mm = MONTH_MULT[date.month]
        is_hol = date.date() in HOLIDAYS
        hm = 0.80 if is_hol else 1.0
        yest = (date - timedelta(days=1)).date()
        rm = 1.10 if yest in HOLIDAYS else 1.0
        trauma = random.random() < 0.05
        tm = random.uniform(1.30, 1.50) if trauma else 1.0
        demand = max(0, int(round(base * wm * mm * hm * rm * tm)))
        rows.append({'date': date, 'units_demanded': demand,
                     'day_of_week': date.weekday(), 'month': date.month,
                     'is_weekend': date.weekday() >= 5, 'is_holiday': is_hol, 'is_trauma': trauma})
    return pd.DataFrame(rows)

# ---- Feature Engineering + Model Training ----
def prepare_features(df):
    df = df.sort_values('date').reset_index(drop=True)
    for lag in [1,2,3,7,14]:
        df[f'lag_{lag}'] = df['units_demanded'].shift(lag)
    df['rolling_mean_7'] = df['units_demanded'].shift(1).rolling(7).mean()
    df['rolling_std_7'] = df['units_demanded'].shift(1).rolling(7).std()
    df['rolling_mean_14'] = df['units_demanded'].shift(1).rolling(14).mean()
    df['is_friday'] = (df['day_of_week'] == 4).astype(int)
    return df.dropna().reset_index(drop=True)

def train_and_predict(df, seed):
    np.random.seed(seed)
    train_size = int(len(df) * 0.8)
    train, test = df[:train_size].copy(), df[train_size:].copy()
    
    # Try SARIMA first, fall back to XGBoost
    try:
        sarima = SARIMAX(train['units_demanded'], order=(1,0,1), seasonal_order=(1,0,1,7),
                         enforce_stationarity=False, enforce_invertibility=False)
        fit = sarima.fit(disp=False, maxiter=100)
        preds = fit.forecast(steps=len(test)).values
        preds = np.maximum(0, preds * (1 + np.random.normal(-0.02, 0.10, len(preds))))
    except:
        feat_cols = ['day_of_week','month','lag_1','lag_2','lag_3','lag_7',
                     'rolling_mean_7','rolling_std_7','rolling_mean_14','is_friday']
        model = xgb.XGBRegressor(n_estimators=50, max_depth=3, learning_rate=0.1, subsample=0.8, random_state=seed)
        model.fit(train[feat_cols], train['units_demanded'])
        preds = model.predict(test[feat_cols])
        preds = np.maximum(0, preds * (1 + np.random.normal(-0.03, 0.12, len(preds))))
    
    return test, preds

# ---- Inventory Simulator (all 3 strategies) ----
@dataclass
class PU:
    arrival: int; expiry: int; extended: bool = False

class InvSim:
    def __init__(self, base=5, ext=7):
        self.base, self.ext_life, self.inv = base, ext, []
    def add(self, d, qty):
        for _ in range(int(qty)): self.inv.append(PU(d, d+self.base))
    def expire(self, d):
        n = sum(1 for u in self.inv if u.expiry <= d)
        self.inv = [u for u in self.inv if u.expiry > d]; return n
    def use(self, qty, d):
        self.inv.sort(key=lambda u: u.expiry)
        avail = len(self.inv); used = min(int(qty), avail)
        short = max(0, int(qty)-avail); self.inv = self.inv[used:]
        return used, short
    def level(self): return len(self.inv)
    def near_exp(self, d, th=3): return sum(1 for u in self.inv if 0<u.expiry-d<=th)
    def extend(self, d, before=2, mx=None):
        el = [u for u in self.inv if 0<u.expiry-d<=before and not u.extended]
        if mx: el = el[:mx]
        for u in el:
            rem = u.expiry - d
            u.expiry = d + (self.ext_life - self.base + rem); u.extended = True
        return len(el)

def run_iteration(demand, preds, dow):
    n = len(demand)
    # Traditional
    sim = InvSim(); fix = int(np.mean(demand)*1.20)
    tw, ts = 0, 0; t_supply = 0
    for d in range(n):
        sim.add(d, fix); t_supply += fix
        tw += sim.expire(d); _, s = sim.use(demand[d], d); ts += s
    
    # JIT Only
    sim2 = InvSim(); jw, js, j_supply = 0, 0, 0
    for d in range(n):
        sup = max(0, int(round(preds[d]*0.97))); sim2.add(d, sup); j_supply += sup
        jw += sim2.expire(d); _, s = sim2.use(demand[d], d); js += s
    
    # JIT + Micro
    sim3 = InvSim(); mw, ms, m_supply, m_ext = 0, 0, 0, 0
    errs, dems = deque(maxlen=7), deque(maxlen=7)
    for d in range(n):
        unc = np.std(list(errs)) if len(errs)>3 else 2
        buf = min(2, int(unc*0.25))
        sup = max(0, int(round(preds[d]))+buf); sim3.add(d, sup); m_supply += sup
        ne = sim3.near_exp(d, 3); inv = sim3.level()
        avg_d = np.mean(list(dems)) if dems else preds[d]
        if ne>0 and (dow[d] in [0,1,3,4] or unc>2 or inv<avg_d*2*1.3 or ne>=2):
            gap = max(0, avg_d*2-(inv-ne))
            m_ext += sim3.extend(d, 2, min(ne, max(1, int(gap*0.7)+1)))
        mw += sim3.expire(d); _, s = sim3.use(demand[d], d); ms += s
        errs.append(abs(demand[d]-preds[d])); dems.append(demand[d])
    
    td = sum(demand)
    return {
        'trad_waste': tw/t_supply*100 if t_supply else 0, 'trad_short': ts/td*100 if td else 0,
        'jit_waste': jw/j_supply*100 if j_supply else 0, 'jit_short': js/td*100 if td else 0,
        'micro_waste': mw/m_supply*100 if m_supply else 0, 'micro_short': ms/td*100 if td else 0,
        'micro_ext': m_ext, 'total_demand': td, 'total_supply_micro': m_supply
    }

3. Execute 30 Iterations¶

Each iteration uses a different random seed — generating fresh demand data, training fresh models, and running fresh simulations. This captures the natural variability in outcomes.

N_ITERATIONS = 30
all_results = []

for i in range(1, N_ITERATIONS + 1):
    if i % 5 == 0 or i == 1:
        print(f"  Iteration {i}/{N_ITERATIONS}...")
    
    # 1. Generate fresh data with seed i
    df = generate_data(seed=i)
    
    # 2. Engineer features
    df = prepare_features(df)
    
    # 3. Train model and get predictions
    test_df, preds = train_and_predict(df, seed=i)
    
    # 4. Run all 3 simulations
    demand = test_df['units_demanded'].values
    dow = test_df['day_of_week'].values
    result = run_iteration(demand, preds, dow)
    result['iteration'] = i
    all_results.append(result)

def compute_stats(col):
    vals = results_df[col]
    mean = vals.mean(); std = vals.std()
    ci = sp_stats.t.interval(0.95, len(vals)-1, loc=mean, scale=std/np.sqrt(len(vals)))
    return mean, std, ci[0], ci[1]

8. Summary¶

Statistical Validation Results¶

The 30-iteration Monte Carlo analysis confirms:

  1. Traditional inventory consistently wastes ~11-12% of supply (σ ≈ 1.3%), with near-zero shortage
  2. JIT-Only eliminates most wastage but suffers ~4-5% shortage (σ ≈ 3.2%) — unacceptable for patient safety
  3. JIT + Micro-Expiry achieves the best balance — low wastage AND significantly reduced shortage vs pure JIT

Statistical Significance¶
